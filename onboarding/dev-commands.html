11. 개발 명령어

# Flutter (항상 fvm 사용)
fvm flutter run                  # 연결된 디바이스/시뮬레이터에서 실행
fvm flutter run --release        # Release build — Rust crypto ~5-10x faster
fvm flutter test                 # Dart 단위 테스트
fvm flutter analyze              # 정적 분석

# Rust
cd rust && cargo test            # Rust 단위 테스트
cd rust && cargo check           # 빠른 컴파일 체크

# FRB codegen (rust/src/api/*.rs 변경 후)
flutter_rust_bridge_codegen generate   # 프로젝트 루트에서 실행!

# iOS 시뮬레이터
./clear-app.sh                   # 시뮬레이터에서 앱 + 키체인 초기화

# Rust 로그 보기 (별도 터미널에서 실행)
# FRB가 Rust log 크레이트 출력을 os_log로 보내므로 flutter run 터미널에 안 보임
log stream --predicate 'subsystem == "frb_user"' --level info
